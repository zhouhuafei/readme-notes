  wx.$globalData.empid = false

    const arr = getCurrentPages().filter(v => v).map(v => v.route)
    console.log('arr--------', arr)
    arr.forEach(v => {
      if (v.indexOf('pages/live-board/index') !== -1 || v.indexOf('pages/index/index') !== -1) {
        wx.$globalData.empid = true
      }
    })
  }, 2000)
