## 暴力递归
> 性能奇差无比，计算极为耗时。40以上的计算容易导致浏览器假死。
```javascript
function fib(n) {
  if(n === 1 || n === 2) {
    return 1
  }
  return fib(n-1) + fib(n-2)
}
console.time('fib')
console.log(fib(6)) // 8
console.log(fib(40)) // 102334155
console.timeEnd('fib') // 1000ms左右
```

## 递归缓存策略
```
fib.cache = []
function fib(n) {
  if(n === 1 || n === 2) {
    return 1
  }
  if(fib.cache[n]) return fib.cache[n]
  fib.cache[n] = fib(n-1) + fib(n-2)
  return fib.cache[n]
}
console.time('fib')
console.log(fib(6)) // 8
console.log(fib(40)) // 102334155
console.timeEnd('fib') // 0.3ms左右
```

## 使用循环
```
function fib(n) {
  let num1= 1,num2= 1,sum;
  for(let i = 3; i <= n; i += 1) {
    sum = num1 + num2
    num1 = num2
    num2 = sum
  }
  return sum
}
console.time('fib')
console.log(fib(6)) // 8
console.log(fib(40)) // 102334155
console.timeEnd('fib') // 0.3ms左右
```
