* null身上没有方法
    - 没有toString方法
    - 也没有valueOf方法
* null > 0; // false
* null < 0; // false
* null == 0; // false
* null == undefined; // true
* null >= 0; // true
* null <= 0; // true

* null <= 1; // true
* null >= 1; // false
* null <= -1; // false
* null >= -1; // true

* null < 1; // true
* null > 1; // false
* null < -1; // false
* null > -1; // true
