```
// 长按事件
var timer = 0 // 初始化起始时间
document.querySelector('.wrap').addEventListener('touchstart', 'p', function (e) {
  e.stopPropagation()
  timer = setTimeout(function () {
    showAlert()
  }, 500) // 这里设置长按响应时间
})

document.querySelector('.wrap').addEventListener('touchend', 'p', function (e) {
  e.stopPropagation()
  clearTimeout(timer)
})

function showAlert () {
  console.log('你长按了')
}
```
