/**
 * 实现函数parse，将服务端返回的公交数据按下面定义的规则优先级输出：
 *
 * 1. 输出常规线路，并按照线路数字从小到大排列(如20路，301路等这些都是常规路线)
 * 2. 输出地铁线路，并按照线路数字从小到大排序
 * 3. 输出其它线路，并按照线路名称长短从小到大排序
 * */

function parse ({ data: { linedetails } }) {
  const arrCategory = [[], [], []]
  Object.keys(linedetails).forEach(key => {
    const { name } = linedetails[key]
    const re1 = name.match(/(\d+)路/) || []
    const re2 = name.match(/地铁(\d+)号线/) || []
    const obj = { key, name }
    if (re1.length) {
      arrCategory[0].push({ ...obj, road: +re1[1] })
    } else if (re2.length) {
      arrCategory[1].push({ ...obj, road: +re2[1] })
    } else {
      arrCategory[2].push({ ...obj, road: name.length })
    }
  })

  // 方式1
  // let arrTemp = []
  // arrCategory.forEach(item => {
  //   arrTemp = arrTemp.concat(item.sort((obj1, obj2) => (obj1.road - obj2.road)))
  // })
  // return arrTemp.map(({ key, name }) => ({ [key]: { name } }))

  // 方式2
  return arrCategory
    .reduce((result, current) => [...result, ...current.sort((obj1, obj2) => (obj1.road - obj2.road))], [])
    .map(({ key, name }) => ({ [key]: { name } }))
}

const arrParse = parse({
  code: 0,
  data: {
    lines: '20路,301路,5路,地铁5号线,机场大巴线,107路,机场快轨',
    lineids: 'lzbd,lwes,lxid,lwic,lwdf,ldfx,loin',
    linedetails: {
      lwdf: { name: '机场大巴线' },
      lwes: { name: '301路' },
      lwic: { name: '地铁5号线' },
      ldfx: { name: '107路' },
      lzbd: { name: '20路' },
      lxid: { name: '5路' },
      loin: { name: '机场快轨' }
    }
  }
})

const arrExpect = [
  { lxid: { name: '5路' } },
  { lzbd: { name: '20路' } },
  { ldfx: { name: '107路' } },
  { lwes: { name: '301路' } },
  { lwic: { name: '地铁5号线' } },
  { loin: { name: '机场快轨' } },
  { lwdf: { name: '机场大巴线' } }
]

console.log('arrParse：', arrParse)
console.log('arrExpect：', arrExpect)
console.log('arrParse和arrExpect是否相同：', JSON.stringify(arrParse) === JSON.stringify(arrExpect))
